const blurBox = document.getElementById("blurBox"); const images = [ "images/thumbring.png","images/knotthumb.png","images/thumbnailegg.jpg", "images/mercurythumbnail.png","images/dreamthumb.jpg","images/t1.jpg", "images/t2.jpg","images/t3.jpg","images/t4.jpg","images/t5.jpg", "images/thumbnut.jpg","images/midtripthumb.jpg","images/mantisthumb.png" ]; let activeImages = []; function spawnImage() { if (activeImages.length >= 2) return; const src = images[Math.floor(Math.random() * images.length)]; const preloaded = new Image(); preloaded.src = src; preloaded.onload = () => { const img = document.createElement("img"); img.src = src; img.style.opacity = 0; const scale = 1.1 + Math.random() * 0.5; img.style.width = `${scale*100}%`; img.style.height = `${scale*100}%`; const extraX = (scale-1) * 300 / 2; const extraY = (scale-1) * 300 / 2; const dx = (Math.random()*2*extraX)-extraX; const dy = (Math.random()*2*extraY)-extraY; img.style.transform = `translate(-50%, -50%) translate(${dx}px,${dy}px)`; blurBox.appendChild(img); activeImages.push(img); // Faster fade-in 20–60s const fadeInDuration = 20000 + Math.random()*40000; img.style.transition = `opacity ${fadeInDuration}ms linear`; setTimeout(() => img.style.opacity = 1, 50); // Blur animation let blurValue = 10 + Math.random()*30; function animateBlur() { blurValue += (Math.random()*0.1-0.05); blurValue = Math.max(10, Math.min(40, blurValue)); img.style.filter = `blur(${blurValue}px)`; requestAnimationFrame(animateBlur); } animateBlur(); // Fade-out after 15–60s const lifeTime = 15000 + Math.random()*45000; setTimeout(() => { const fadeOutDuration = 20000 + Math.random()*40000; img.style.transition = `opacity ${fadeOutDuration}ms linear`; img.style.opacity = 0; setTimeout(() => { if (blurBox.contains(img)) blurBox.removeChild(img); activeImages.splice(activeImages.indexOf(img),1); if(activeImages.length < 2) spawnImage(); }, fadeOutDuration); }, lifeTime); // Drift within allowed extra pixels function driftImg() { const dxNew = (Math.random()*2*extraX)-extraX; const dyNew = (Math.random()*2*extraY)-extraY; const duration = 15000 + Math.random()*25000; img.style.transition = `transform ${duration}ms linear`; img.style.transform = `translate(-50%, -50%) translate(${dxNew}px,${dyNew}px)`; setTimeout(driftImg, duration); } driftImg(); }; } // Start with 1–2 images spawnImage(); setTimeout(spawnImage, 10000 + Math.random()*10000); // Periodically try to spawn new image faster setInterval(() => { if(activeImages.length < 2) spawnImage(); }, 20000 + Math.random()*20000);
